1. A* Algorithm
#include<iostream>
#include<queue>
using namespace std;

struct Node{
    int x,y,cost;
};

int main(){

    int grid[3][3] = {
        {0,0,0},
        {1,0,1},
        {0,0,0}
    };

    queue<Node> q;

    bool visited[3][3] = {false};

    q.push({0,0,0});

    visited[0][0] = true;

    int dx[] = {-1,1,0,0};
    int dy[] = {0,0,-1,1};

    while(!q.empty()){

        Node cur = q.front();
        q.pop();

        cout<<cur.x<<" "<<cur.y<<endl;

        if(cur.x==2 && cur.y==2){
            cout<<"Goal Found\n";
            break;
        }

        for(int i=0;i<4;i++){

            int nx = cur.x + dx[i];
            int ny = cur.y + dy[i];

            if(nx>=0 && ny>=0 &&
               nx<3 && ny<3 &&
               grid[nx][ny]==0 &&
               !visited[nx][ny]){

                visited[nx][ny]=true;

                q.push({nx,ny,cur.cost+1});
            }
        }
    }
}
2. Selection Sort
#include<iostream>
using namespace std;

int main(){

    int arr[] = {64,25,12,22,11};

    int n = 5;

    for(int i=0;i<n-1;i++){

        int min = i;

        for(int j=i+1;j<n;j++){

            if(arr[j]<arr[min]){
                min = j;
            }
        }

        swap(arr[i],arr[min]);
    }

    for(int i=0;i<n;i++){
        cout<<arr[i]<<" ";
    }
}
3. Minimum Spanning Tree (Prim)
#include<iostream>
using namespace std;

int main(){

    int V=5;

    int graph[5][5] = {

        {0,2,0,6,0},
        {2,0,3,8,5},
        {0,3,0,0,7},
        {6,8,0,0,9},
        {0,5,7,9,0}
    };

    bool selected[5]={false};

    selected[0]=true;

    int edgeCount=0;

    while(edgeCount<V-1){

        int min=999;
        int x=0,y=0;

        for(int i=0;i<V;i++){

            if(selected[i]){

                for(int j=0;j<V;j++){

                    if(!selected[j] &&
                       graph[i][j]){

                        if(min>graph[i][j]){

                            min=graph[i][j];

                            x=i;
                            y=j;
                        }
                    }
                }
            }
        }

        cout<<x<<" - "<<y
            <<" : "<<graph[x][y]
            <<endl;

        selected[y]=true;

        edgeCount++;
    }
}
4. Single Source Shortest Path (Dijkstra)
#include<iostream>
using namespace std;

int main(){

    int V=5;

    int graph[5][5] = {

        {0,10,0,5,0},
        {10,0,1,2,0},
        {0,1,0,0,4},
        {5,2,0,0,2},
        {0,0,4,2,0}
    };

    int dist[5];

    bool visited[5];

    for(int i=0;i<V;i++){

        dist[i]=999;

        visited[i]=false;
    }

    dist[0]=0;

    for(int count=0;count<V-1;count++){

        int min=999;
        int u;

        for(int i=0;i<V;i++){

            if(!visited[i] &&
               dist[i]<min){

                min=dist[i];
                u=i;
            }
        }

        visited[u]=true;

        for(int v=0;v<V;v++){

            if(graph[u][v] &&
               !visited[v] &&
               dist[u]+graph[u][v]<dist[v]){

                dist[v]=
                dist[u]+graph[u][v];
            }
        }
    }

    for(int i=0;i<V;i++){

        cout<<"0 -> "<<i
            <<" = "<<dist[i]
            <<endl;
    }
}
5. Job Scheduling Problem
#include<iostream>
#include<vector>
#include<algorithm>

using namespace std;

struct Job{
    char id;
    int deadline;
    int profit;
};

bool cmp(Job a,Job b){

    return a.profit>b.profit;
}

int main(){

    vector<Job> jobs = {

        {'A',2,100},
        {'B',1,19},
        {'C',2,27},
        {'D',1,25},
        {'E',3,15}
    };

    sort(jobs.begin(),
         jobs.end(),
         cmp);

    char slot[4]={'-'};

    int profit=0;

    for(auto job:jobs){

        for(int j=job.deadline;
            j>0;
            j--){

            if(slot[j]=='-'){

                slot[j]=job.id;

                profit+=job.profit;

                break;
            }
        }
    }

    for(int i=1;i<=3;i++){

        cout<<slot[i]<<" ";
    }

    cout<<"\nProfit = "
        <<profit;
}

7. Kruskal Algorithm
#include<iostream>
#include<vector>
#include<algorithm>

using namespace std;

struct Edge{

    int u,v,w;
};

bool cmp(Edge a,Edge b){

    return a.w<b.w;
}

vector<int> parent;

int findParent(int node){

    if(parent[node]==node)
        return node;

    return parent[node]=
           findParent(parent[node]);
}

void unionSet(int u,int v){

    u=findParent(u);
    v=findParent(v);

    parent[u]=v;
}

int main(){

    int V=4;

    vector<Edge> edge={

        {0,1,3},
        {0,2,2},
        {0,3,5},
        {1,3,4},
        {2,3,3}
    };

    parent.resize(V);

    for(int i=0;i<V;i++){
        parent[i]=i;
    }

    sort(edge.begin(),
         edge.end(),
         cmp);

    for(auto e:edge){

        if(findParent(e.u)!=
           findParent(e.v)){

            cout<<e.u
                <<" - "
                <<e.v
                <<" : "
                <<e.w
                <<endl;

            unionSet(e.u,e.v);
        }
    }
}
8. N Queen Problem
#include<iostream>
using namespace std;

int board[4][4]={0};

bool isSafe(int row,int col){

    for(int i=0;i<row;i++){

        if(board[i][col])
            return false;
    }

    for(int i=row,j=col;
        i>=0 && j>=0;
        i--,j--){

        if(board[i][j])
            return false;
    }

    for(int i=row,j=col;
        i>=0 && j<4;
        i--,j++){

        if(board[i][j])
            return false;
    }

    return true;
}

bool solve(int row){

    if(row>=4)
        return true;

    for(int col=0;col<4;col++){

        if(isSafe(row,col)){

            board[row][col]=1;

            if(solve(row+1))
                return true;

            board[row][col]=0;
        }
    }

    return false;
}

int main(){

    solve(0);

    for(int i=0;i<4;i++){

        for(int j=0;j<4;j++){

            cout<<board[i][j]
                <<" ";
        }

        cout<<endl;
    }
}
